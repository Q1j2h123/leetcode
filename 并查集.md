### [1971. 寻找图中是否存在路径](https://leetcode.cn/problems/find-if-path-exists-in-graph/description)

```c
class Solution {
public:
    vector<int>parent;
    bool validPath(int n, vector<vector<int>>& edges, int source, int destination) {
        if (source == destination) return true;
        parent.resize(n);
        for(int i=0;i<n;i++){
            parent[i]=i;
        }
        for(auto& edge:edges){
            int x=edge[0];
            int y=edge[1];
            int rootX=find(x);
            int rootY=find(y);
            if(rootX!=rootY){
                parent[rootY]=rootX;
            }
        }
        if(find(destination)==find(source))return true;
        return false;
    }
    int find(int x){
        if(parent[x]!=x){
            parent[x]=find(parent[x]);
        }
        return parent[x];
    }
};

```


### [冗余连接](https://leetcode.cn/problems/redundant-connection/description)
```c
class Solution {
public:
 vector<int>parent;
    vector<int> findRedundantConnection(vector<vector<int>>& edges) {
        int n=edges.size();
        parent.resize(n+1);
     
      for(int i=1;i<=n;i++){
        parent[i]=i;
      }
    for(auto edge:edges){
        int x=edge[0];
        int y=edge[1];
        int rootX=find(x);
        int rootY=find(y);
        if(rootX==rootY){
            return edge;
        }
        parent[rootY]=rootX;
    }
    return vector<int>();
    }
    int find(int x){
        if(parent[x]!=x){
            parent[x]=find(parent[x]);
        }
        return parent[x];
    }
};

```

### [547.省份数量](https://leetcode.cn/problems/number-of-provinces/description/)

```c

class Solution {
public:
    vector<int>parent;
    int findCircleNum(vector<vector<int>>& isConnected) {
        int n=isConnected.size();
        parent.resize(n);
        for(int i=0;i<n;i++){
            parent[i]=i;
        }
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                if(isConnected[i][j]==1){
                    int rootI=find(i);
                    int rootJ=find(j);
                    if(rootI!=rootJ){
                            parent[rootJ]=rootI;
                    }
                   
                }
            }
        }
        int res=0;
        for(int i=0;i<n;i++){
            if(find(i)==i){
                res++;
            }
        }
        return res;
    }
    int find(int x){
        if(parent[x]!=x){
            parent[x]=find(parent[x]);
        }
        return parent[x];
    }
};
```
